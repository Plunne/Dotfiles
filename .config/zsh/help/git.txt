gs   : git status
ga   : git add <FILES>

gp   : git push
gpp  : git push origin <BRANCH>

gl   : git pull
gu   : git commit -m <MESSAGE> && git push
guu  : git commit -m <MESSAGE> && git push origin <BRANCH>
gc   : git clone <REPOSITORY>
gb   : git checkout <BRANCH>
gbb  : git checkout -b <NEW_BRANCH>
